Storing Data in Arrays

What is an Array?

An array is a collection of variables stored under one name, accessed by index number.

// Declare an array to hold 19 distance readings (0°, 10°, 20°... 180°) int distances[19]; // Store a value at index 0 (angle 0°) distances[0] = 45; // 45 cm at 0 degrees // Store a value at index 9 (angle 90°) distances[9] = 120; // 120 cm at 90 degrees // Read a value int center = distances[9]; // Gets 120

Filling the Scan Array

#include <Servo.h> const int NUM_READINGS = 19; // 0° to 180° in 10° steps int distances[NUM_READINGS]; // Array to store readings Servo scanServo; const int servoPin = 4; const int trigPin = 2; const int echoPin = 3; void performScan() { int index = 0; // Scan from 0° to 180° in 10° steps for (int angle = 0; angle <= 180; angle += 10) { scanServo.write(angle); delay(50); // Wait for servo distances[index] = getDistance(); index++; } } // After scan: distances[0] = reading at 0° // distances[9] = reading at 90° // distances[18] = reading at 180°

Index to Angle Conversion

angle = index × 10    |    index = angle ÷ 10

Finding the Nearest Obstacle

The Algorithm

  1. Assume the first reading is the minimum
  2. Loop through all readings
  3. If current reading is smaller, update minimum
  4. Track the index where minimum was found
  5. Convert index to angle

void findNearest(int* minDist, int* minAngle) { *minDist = distances[0]; int minIndex = 0; for (int i = 1; i < NUM_READINGS; i++) { if (distances[i] < *minDist && distances[i] > 0) { *minDist = distances[i]; minIndex = i; } } *minAngle = minIndex * 10; // Convert index to angle } // Usage: int nearestDist, nearestAngle; findNearest(&nearestDist, &nearestAngle); Serial.print("Nearest: "); Serial.print(nearestDist); Serial.print("cm at "); Serial.print(nearestAngle); Serial.println("°");

Finding the Clearest Path

void findClearest(int* maxDist, int* maxAngle) { *maxDist = distances[0]; int maxIndex = 0; for (int i = 1; i < NUM_READINGS; i++) { if (distances[i] > *maxDist) { *maxDist = distances[i]; maxIndex = i; } } *maxAngle = maxIndex * 10; // Convert index to angle } // Usage: int clearestDist, clearestAngle; findClearest(&clearestDist, &clearestAngle); Serial.print("Clearest path: "); Serial.print(clearestDist); Serial.print("cm at "); Serial.print(clearestAngle); Serial.println("°");

Why Find the Maximum?

The direction with the largest distance reading has the most open space - that's where the robot should go!

Making Navigation Decisions

Robot Field of View:

         90° (straight ahead)
              ↑
              │
    ←─────────┼─────────→
  0°          │          180°
  (right)     │          (left)

Decision Logic:
  • clearestAngle < 80°  → Turn RIGHT
  • clearestAngle > 100° → Turn LEFT
  • else                 → Go FORWARD
void makeDecision(int clearestAngle, int clearestDist) { Serial.print("Decision: "); if (clearestDist < 15) { // Too close everywhere - back up! Serial.println("REVERSE - too close!"); } else if (clearestAngle < 80) { Serial.println("TURN RIGHT"); } else if (clearestAngle > 100) { Serial.println("TURN LEFT"); } else { Serial.println("GO FORWARD"); } }

Complete Scanner Code

#include <Servo.h> const int NUM_READINGS = 19; int distances[NUM_READINGS]; Servo scanServo; const int servoPin = 4, trigPin = 2, echoPin = 3; void setup() { Serial.begin(115200); scanServo.attach(servoPin); pinMode(trigPin, OUTPUT); pinMode(echoPin, INPUT); scanServo.write(90); delay(500); } void loop() { performScan(); printScanData(); int minDist, minAngle, maxDist, maxAngle; findNearest(&minDist, &minAngle); findClearest(&maxDist, &maxAngle); Serial.print("Nearest: "); Serial.print(minDist); Serial.print("cm at "); Serial.print(minAngle); Serial.println("°"); Serial.print("Clearest: "); Serial.print(maxDist); Serial.print("cm at "); Serial.print(maxAngle); Serial.println("°"); makeDecision(maxAngle, maxDist); Serial.println("-------------------"); scanServo.write(90); // Return to center delay(2000); }
