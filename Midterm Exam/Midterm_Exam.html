Code Analysis - Read Carefully
Study the following Arduino code, then answer Questions 31-35.
// Robot Motion Control System

int leftMotorPin = 9;
int rightMotorPin = 10;
int buttonPin = 7;
int potPin = A0;

int speed = 0;
boolean isRunning = false;

void setup() {
  pinMode(leftMotorPin, OUTPUT);
  pinMode(rightMotorPin, OUTPUT);
  pinMode(buttonPin, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop() {
  if (digitalRead(buttonPin) == LOW) {
    isRunning = !isRunning;
    delay(300);
  }

  if (isRunning) {
    speed = analogRead(potPin);
    speed = map(speed, 0, 1023, 100, 255);

    analogWrite(leftMotorPin, speed);
    analogWrite(rightMotorPin, speed);

    Serial.print("Motors ON - Speed: ");
    Serial.println(speed);
  } else {
    analogWrite(leftMotorPin, 0);
    analogWrite(rightMotorPin, 0);
    Serial.println("Motors OFF");
  }

  delay(100);
}
